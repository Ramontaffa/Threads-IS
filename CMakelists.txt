# Define the minimum CMake version required
cmake_minimum_required(VERSION 3.15)

# Name your project and specify that it uses C++
project(AUDIO-THREADS CXX)

# Require C++17 standard (great for std::atomic and std::thread)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

option(AUDIO_THREADS_ENABLE_GUI "Build with Dear ImGui + GLFW UI" ON)

# -------------------------------------------------------------------
# 1. Fetch and configure RtAudio automatically
# -------------------------------------------------------------------
include(FetchContent)

message(STATUS "Downloading and configuring RtAudio...")
FetchContent_Declare(
    rtaudio
    GIT_REPOSITORY https://github.com/thestk/rtaudio.git
    GIT_TAG        6.0.1 # Use a stable release version
)
# This command downloads RtAudio and makes its targets available to us
FetchContent_MakeAvailable(rtaudio)

if(AUDIO_THREADS_ENABLE_GUI)
    message(STATUS "Downloading and configuring GUI dependencies (GLFW + Dear ImGui)...")

    if(UNIX AND NOT APPLE)
        set(GLFW_BUILD_WAYLAND OFF CACHE BOOL "" FORCE)
        set(GLFW_BUILD_X11 ON CACHE BOOL "" FORCE)
    endif()

    FetchContent_Declare(
        glfw
        GIT_REPOSITORY https://github.com/glfw/glfw.git
        GIT_TAG        3.4
    )

    FetchContent_Declare(
        imgui
        GIT_REPOSITORY https://github.com/ocornut/imgui.git
        GIT_TAG        v1.91.9b
    )

    FetchContent_MakeAvailable(glfw imgui)
    find_package(OpenGL REQUIRED)

    add_library(imgui_lib STATIC
        ${imgui_SOURCE_DIR}/imgui.cpp
        ${imgui_SOURCE_DIR}/imgui_draw.cpp
        ${imgui_SOURCE_DIR}/imgui_tables.cpp
        ${imgui_SOURCE_DIR}/imgui_widgets.cpp
        ${imgui_SOURCE_DIR}/backends/imgui_impl_glfw.cpp
        ${imgui_SOURCE_DIR}/backends/imgui_impl_opengl3.cpp
    )

    target_include_directories(imgui_lib
        PUBLIC
        ${imgui_SOURCE_DIR}
        ${imgui_SOURCE_DIR}/backends
    )

    target_link_libraries(imgui_lib
        PUBLIC
        glfw
        OpenGL::GL
    )
endif()

# -------------------------------------------------------------------
# 2. Define our executable and its source files
# -------------------------------------------------------------------
add_executable(AUDIO-THREADS
    src/main.cpp
    src/AudioEngine.cpp
    src/AudioSetup.cpp
    src/UIManager.cpp
    src/StatusMonitor.cpp
    src/Visualizador.cpp
)

if(AUDIO_THREADS_ENABLE_GUI)
    target_sources(AUDIO-THREADS PRIVATE src/GuiManager.cpp)
    target_compile_definitions(AUDIO-THREADS PRIVATE AUDIO_THREADS_ENABLE_GUI=1)
endif()

# -------------------------------------------------------------------
# 3. Tell the compiler where to find our .h header files
# -------------------------------------------------------------------
target_include_directories(AUDIO-THREADS PRIVATE include)

# -------------------------------------------------------------------
# 4. Link the libraries to our executable
# -------------------------------------------------------------------
# Find the system's threading library (required for std::thread)
find_package(Threads REQUIRED)

# Link RtAudio and the Threading library to our executable
target_link_libraries(AUDIO-THREADS 
    PRIVATE 
    rtaudio
    Threads::Threads
)

if(AUDIO_THREADS_ENABLE_GUI)
    target_link_libraries(AUDIO-THREADS PRIVATE imgui_lib)
endif()
# Copy the songs folder to the build directory during configuration
file(COPY songs DESTINATION ${CMAKE_CURRENT_BINARY_DIR})
